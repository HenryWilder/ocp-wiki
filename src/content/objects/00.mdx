export const randomLengthStr = (minWordCount, maxWordCount, minWordLength, maxWordLength) => {
    let str = '';
    const words = Math.floor(Math.random() * (maxWordCount - minWordCount) + minWordCount);
    for (let i = 0; i < words; i++)
        str += ('x'.repeat(Math.floor(Math.random() * (maxWordLength - minWordLength) + minWordLength))+' ')
    return str;
}
